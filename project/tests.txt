1) Массив:

# вставки
./main --file array.txt --query "MPUSH myarray 10"
./main --file array.txt --query "MPUSH myarray 20"
./main --file array.txt --query "MPUSH myarray 30"
./main --file array.txt --query "PRINTM myarray"

# вставка по индексу (в начало, в середину, в конец)
./main --file array.txt --query "MINDEX myarray 0 5"    # вставить 5 в начало
./main --file array.txt --query "MINDEX myarray 2 15"   # вставить 15 в середину
./main --file array.txt --query "MINDEX myarray 6 40"   # вставка в конец (если индекс==size)

./main --file array.txt --query "PRINTM myarray"

# чтение по индексу корректный и неверный
./main --file array.txt --query "MGET myarray 2"
./main --file array.txt --query "MGET myarray 999"      # ожидаем сообщение об ошибке / -1

# замена и удаление
./main --file array.txt --query "MSET myarray 2 99"
./main --file array.txt --query "MDEL myarray 0"
./main --file array.txt --query "PRINTM myarray"

2) Красно-чёрное дерево:

# вставки
./main --file data.txt --query "TINSERT mytree 50"
./main --file data.txt --query "TINSERT mytree 30"
./main --file data.txt --query "TINSERT mytree 70"
./main --file data.txt --query "TINSERT mytree 20"
./main --file data.txt --query "TINSERT mytree 40"
./main --file data.txt --query "TINSERT mytree 60"
./main --file data.txt --query "TINSERT mytree 80"
./main --file data.txt --query "PRINT mytree"

# поиск
./main --file data.txt --query "TGET mytree 60"   # TRUE
./main --file data.txt --query "TGET mytree 99"   # FALSE

# удаление листа, узла с одним ребёнком, узла с двумя детьми
./main --file data.txt --query "TDEL mytree 20"   # leaf
./main --file data.txt --query "TDEL mytree 30"   # may have one child after previous deletions
./main --file data.txt --query "TDEL mytree 50"   # root with two children
./main --file data.txt --query "PRINT mytree"

# удаление несуществующего
./main --file data.txt --query "TDEL mytree 999"

3) Односвязный список:

# вставки
./main --file flist.txt --query "FPUSH mylist 10"
./main --file flist.txt --query "FPUSH mylist 30"
./main --file flist.txt --query "FPUSHFRONT mylist 5"
./main --file flist.txt --query "PRINTF mylist"

# вставки до/после
./main --file flist.txt --query "FINSERTBEFORE mylist 30 20"
./main --file flist.txt --query "FINSERTAFTER mylist 10 15"
./main --file flist.txt --query "PRINTF mylist"

# чтение
./main --file flist.txt --query "FGET mylist 2"
./main --file flist.txt --query "FGET mylist 999"   # out of range

# contains and delete
./main --file flist.txt --query "FCONTAINS mylist 20"
./main --file flist.txt --query "FDEL mylist 20"
./main --file flist.txt --query "FDEL mylist 999"   # удалить несуществующее
./main --file flist.txt --query "PRINTF mylist"

4) Двусвязный список:

# вставки
./main --file llist.txt --query "LPUSH mylist 100"
./main --file llist.txt --query "LPUSH mylist 300"
./main --file llist.txt --query "LPUSHFRONT mylist 50"
./main --file llist.txt --query "PRINTL mylist"

./main --file llist.txt --query "LINSERTBEFORE mylist 300 200"
./main --file llist.txt --query "LINSERTAFTER mylist 100 75"
./main --file llist.txt --query "PRINTL mylist"

./main --file llist.txt --query "LGET mylist 0"
./main --file llist.txt --query "LCONTAINS mylist 75"
./main --file llist.txt --query "LDEL mylist 50"    # удаление головы
./main --file llist.txt --query "LDEL mylist 300"   # удаление хвоста
./main --file llist.txt --query "PRINTL mylist"
